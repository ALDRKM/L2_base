function longest(arr, n) {
  let str = [];
  for (let i = 0; i < arr.length; i++) {
    str.push(arr[i]);
  }
  for (let i = 0; i < str.length - 1; i++) {
    for (let j = 0; j < str.length - 1 - i; j++) {
      if (str[j].length < str[j + 1].length) {
        let temp = str[j];
        str[j] = str[j + 1];
        str[j + 1] = temp;
      }
    }
  }
  return str[n - 1];
}
console.log(longest(['Hello','World','Codewars','Katas'], 3));
